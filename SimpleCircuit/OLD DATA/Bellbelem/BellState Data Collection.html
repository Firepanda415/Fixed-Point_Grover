def counts_to_probs(counts):
    counts_shots = np.sum([v for v in counts.values()])
    probs = np.zeros(2**(n), dtype=np.float64)
    for k,v in counts.items():
        probs[int(k,2)] = v/counts_shots
    return probs

def VD(p,q):
    np_p = np.array(p)
    np_q = np.array(q)
    return 0.5*np.abs(np_p - np_q).sum()

def one_itr(qc):
    qc.h(0)
    qc.cx(0,1)
    qc.barrier()
    qc.cx(0,1)
    qc.h(0)
    qc.barrier()

# Bell0 state
def pure_circuit(name=None):
    r = QuantumRegister(n)
    qc = QuantumCircuit(r, name=name)
    for it in range(itr):
        one_itr(qc)
    return r, qc  

def noise_gates_circuit(noise_params):
    angles = noise_params.reshape((3, n)) #NOT A NUMPY ARRAY FROM SCIPY.OPTIMIZE
    
    r = QuantumRegister(n)
    qc = QuantumCircuit(r)
    for it in range(itr):
        one_itr(qc)
        for q in range(n):
            qc.u(angles[0,n-q-1], angles[1,n-q-1], angles[2,n-q-1], n-q-1) # Add noise/denoise here
        qc.barrier()
    return r, qc

circs = []

    print("\n\nNumber of Iterations = {:d}".format(itr))
    r, simple_qc = pure_circuit("BellStateItr{:d}".format(itr))
    simple_state = Statevector.from_instruction(simple_qc)
    # Check state
    print("Ideal State", simple_state.data)
    # Measure
    simple_qc.measure_all()
    simple_qc.draw('mpl',filename=DataFileAddress+'bell_circ_itr_{:d}_depth_{:d}.pdf'.format(itr, simple_qc.depth()))
    print("")
    # Transpile circuit
    trans_simple_qc = transpile(simple_qc, backend, initial_layout = interested_qubits, optimization_level=0)
    trans_simple_qc.draw('mpl',filename=DataFileAddress+'bell_circ_trans_itr_{:d}_depth_{:d}.pdf'.format(itr, trans_simple_qc.depth()))
    print("Original Depth = {:d}, Transpiled Depth = {:d}".format(simple_qc.depth(), trans_simple_qc.depth()))
    circs.append(trans_simple_qc)
